import numpy as np
import matplotlib.pyplot as plt
import random

# 
N = 100  # 
p_i = 0.01  # 
p_r = 0.0  # 
p_s = 1 - p_i - p_r  # 
beta = 0.3  # 
gamma = 0.1  # 
mu = 0.05  # 

# 
state = np.zeros((N, N))
for i in range(N):
    for j in range(N):
        if random.random() < p_i:
            state[i, j] = 1  # 
        elif random.random() < p_r:
            state[i, j] = 2  # 
        else:
            state[i, j] = 0  # 

# 
def update(state):
    new_state = np.zeros((N, N))
    for i in range(N):
        for j in range(N):
            if state[i, j] == 0:  # 
                infected_neighbors = np.sum(state[max(i-1,0):min(i+2,N), max(j-1,0):min(j+2,N)] == 1)
                if random.random() < 1 - (1 - beta) ** infected_neighbors:
                    new_state[i, j] = 1  # 
            elif state[i, j] == 1:  # 
                if random.random() < gamma:
                    new_state[i, j] = 2  # 
                else:
                    new_state[i, j] = 1  # 
            else:  # 
                if random.random() < mu:
                    new_state[i, j] = 0  # 
                else:
                    new_state[i, j] = 2  # 
    return new_state

# 
T = 100
for t in range(T):
    state = update(state)

# 
plt.imshow(state, cmap='viridis')
plt.show()

import networkx as nx
import numpy as np
import matplotlib.pyplot as plt
import random

# 
N = 1000  # 
p_i = 0.01  # 
p_r = 0.0  # 
p_s = 1 - p_i - p_r  # 
beta = 0.3  # 
gamma = 0.1  # 
mu = 0.05  # 

# 
G = nx.erdos_renyi_graph(N, 0.1)

# 
state = np.zeros(N)
for i in range(N):
    if random.random() < p_i:
        state[i] = 1  # 
    elif random.random() < p_r:
        state[i] = 2  # 
    else:
        state[i] = 0  # 

# 
def update(state):
    new_state = np.zeros(N)
    for i in range(N):
        if state[i] == 0:  # 
            infected_neighbors = np.sum([state[j] == 1 for j in G.neighbors(i)])
            if random.random() < 1 - (1 - beta) ** infected_neighbors:
                new_state[i] = 1  # 
        elif state[i] == 1:  # 
            if random.random() < gamma:
                new_state[i] = 2  # 
            else:
                new_state[i] = 1  # 
        else:  # 
            if random.random() < mu:
                new_state[i] = 0  # 
            else:
                new_state[i] = 2  # 
    return new_state

# 
T = 100
for t in range(T):
    state = update(state)

# 
pos = nx.spring_layout(G)
colors = ['blue' if s == 0 else 'red' if s == 1 else 'green' for s in state]
nx.draw(G, pos, node_color=colors, with_labels=False)
plt.show()
