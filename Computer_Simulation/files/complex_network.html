%初始化网络数据， N 个节点，最近邻的 2m 个节点相互连边
N = 10 ; m = 2;
p = 0.3;%重连接概率 p
A = zeros(N);
%建立初始的环状的规则网络， 写邻接矩阵 A
for	i = 1 : N - m
    for	j = i + 1: i + m
        A( i , j) = 1 ; A( j , i) = 1 ;
    end
end
for	i = N - m +1:N
    for  j = [ i + 1 :N ,   1  : i -   N  +   m]
        A( i , j ) = 1 ; A(j , i ) = 1;
    end
end

function SWplot(A)    % 根据邻接矩阵画出该小世界网络的图形
    len = size(A,1);  
    rho = 10 ;	% 限制图尺寸的大小
    theta = 0:( 2 * pi / len ) : 2 * pi*( 1 - 1 / len ) ;
    [px , py] = pol2cart(theta' , rho) ;
    gplot(A, [px, py], '-ok');    axis off; hold on; 
    text(px, py + 0.3, num2str((1:len)'),'FontSize',14);  
    hold off;
end

for	i = 1 :N - m
    for	j = i + 1 : i + m
        r = rand(1);
        if r <=p % 随机选取一个数
            unconect = find(A(i,:) ==0) ;	% 取出邻接矩阵中的非 o 元素位置
            Q = length(unconect);	% 求出非 o 元素个数
            k = ceil(Q*rand(1)) ;	% 向上取整
            A(i,unconect(k)) = 1;
            A(unconect(k),i) = 1 ;	% 连接这一对点
            A(i,j) = 0;  A(j,i)=0;	% 断开原来的边
        end
    end
end

function [G, xy] = graph(N, network)
    G = zeros(N);
    [in ,out]=  find( network) ;
    for k = 1 : numel(in)   
        G(in(k),out(k)) = 1;
    end
    G = sparse(G ) ;
    x = round(rand(1,N)*N);
    y = round(rand(1,N)*N);
    xy = [ x', y'];
end

function p = updateDist(p,vertex,numDegrees)
    totalDegrees = sum(numDegrees );
    for	i = 1: vertex
        p(i) = numDegrees(i)/totalDegrees;
    end
end
