function M = Ising(T)    % T is the tempure.  +-+-+   Ising area.
    nTrials = 10000;    % for stat.          |3|4|   N = 2.
    startup = 3000;    % for steady.         +-+-+   free boundary.
    d = zeros(1, nTrials);   % results.       |1|2| 
    beta = 1/T;              %                +-+-+ 
    M = 0;
    s = 2 * round(rand(1, 4)) - 1;   % randomly set 1 or -1.
    Es = s(1) * s(2) + s(3) * s(4);        
    Es = Es + s(1) * s(3) + s(2) * s(4);   
    Es = - Es;                              
    for t = 1:nTrials        
        k = fix(1 + 4 * rand);  % randomly pick 1 ~ 4.
        y = s;         % y is the suggestion dist in s' neighbour. 
        y(k) = -s(k);  % randomly flip the status on one point.
        Ey = y(1) * y(2) + y(3) * y(4);
        Ey = Ey + y(1) * y(3) + y(2) * y(4);
        Ey = - Ey;
        h = min(1, exp(-beta * (Ey - Es)));
        if (rand < h)
            s = y;
        end
        if (t > startup)
            Ms = s(1) + s(2) + s(3) + s(4);  % cheating! 
            M = M + Ms;                      % this is not Boltzmann dist.
            d(t - startup) = Ms;
        end
    %    if(mod(t, 50000)==0) fprintf('Round %d ...\n', t); end
    end
 %   x = -4:1:4;
 %   hist(d, x);
    M = (M / nTrials) / 4;
end

for k = 1:Ncase
    T = x(k);
    E = 0;
    S  =  2*(randi([0,1], Nsize, Nsize) - 0.5);
    fprintf("Start simulation for case %d, where T = %d ... \n", k, T); 
    for t = 1:NMaxSim
        if(mod(t, n_verbose)==0) fprintf("    Try #%d ... \n", t); end
        
        i = randi([2,Nsize-1]);
        j = randi([2,Nsize-1]);
        a = [i-1,j;i+1,j;i,j-1;i,j+1];
        %
        b = randi([1,4]);
        c = a(b,:);
        i1 = c(1);
        j1 = c(2);
        
        %% Try to fix these with Metropis Algorithm
        Energybefore = -J*S(i,j)*S(i1,j1) - H*S(i,j)*S(i1,j1);
        Energyafter = J*S(i,j)*S(i1,j1) + H*S(i,j)*S(i1,j1);
        Energydifference = Energyafter - Energybefore;
        
        p = 1/(1 + exp(Energydifference/T));  % 这个分布比较关键
        
        if (Energydifference < 0)
            S(i,j) = -S(i,j);
            e = p*Energyafter + (1-p)* Energybefore;%  
        else
            S(i,j) = S(i,j);
            e = Energybefore;
        end

        %%
        if(t > N0) E = E + e/(NMaxSim - N0);  end        
%        if(mod(t, n_verbose)==0) imagesc(S); end
    end
%   save the results
    Energy(k) = E;
    Image(:,:,k) = S;
    fprintf("==== Simulation Finished for case %d with E = %f. ====\n\n", k, E); 
end
